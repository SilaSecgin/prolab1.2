void XOR (int a, int b){

    else c=1;

void AND (int a, int b){

    int c=0;
    if (a==1 && b==1) c=1;
    else c=0;

    return c;

}
void OR (int a, int b){

    int c=0;
    if (a==0 && b==0) c=0;
    else c=1;

    return c;
}

void NAND (int a, int b){

    int c=0;
    if (a==1 && b==1) c=0;
    else c=1;

    return c;

}

void NOR (int a, int b){

    int c=0;
    if (a==1 && b==1) c=1;
    else c=0;

    return c;

}

void NOT(int a){

    int c=0;
    if (a==0 ) c=1;
    else if (a==1) c=0;

    return c;

}
